% ── Abstract ──────────────────────────────────────────────────────────────────
\begin{block}{Abstract}
Autonomous AI agents powered by Large Language Models (LLMs) are increasingly used for
offensive security tasks. When multiple LLM-driven agents interact with a target system,
defenders currently cannot determine \textbf{which model is behind each session}.
TRACE fingerprints the \textbf{shell command patterns} each LLM family produces ---
achieving \textbf{98\% cross-validated accuracy} across 7 model families and 3 agent
scaffolds. We show these fingerprints enable \textbf{targeted Defensive Prompt Injection},
routing the highest-ASR payload to each identified family.
\end{block}

% ── Problem ───────────────────────────────────────────────────────────────────
\begin{alertblock}{Problem}
\begin{itemize}
  \item Frontier LLMs now autonomously hack systems --- agents run \texttt{nmap}, \texttt{gobuster},
        exploit CVEs, and pivot laterally \textbf{without human intervention}
  \item Defenders see only shell commands --- the LLM identity is \textbf{hidden behind the scaffold}
  \item Each LLM family responds differently to prompt injection defenses
  \item Without fingerprinting, defenders must deploy \textbf{one-size-fits-all} countermeasures
\end{itemize}
\textbf{Goal:} Identify the LLM family from terminal behavior alone, then deploy \textbf{targeted} defenses.
\end{alertblock}

% ── Method summary ────────────────────────────────────────────────────────────
\begin{block}{Classification Pipeline --- From Failure to TF-IDF}

\textbf{What didn't work (TRACE v1):} 80 hand-crafted behavioral features ---
OIPS (deliberation depth), error recovery rate, command mutual information, phase
transition entropy. \textbf{Result: LOSO F1 = 0.11.} The features predicted
\textit{scaffold}, not \textit{model}. PentestGPT's rigid turn structure, ReAct's
reasoning-action framing, CC's tool patterns --- all leaked into features.

\vspace{0.3em}
\textbf{What works (TRACE v2):} Treat each session as a text document.

\begin{table}
\centering
\begin{tabular}{l l}
\toprule
\textbf{Stage} & \textbf{Details} \\
\midrule
Input & Session commands $\rightarrow$ single text string \\
Vectorizer & TF-IDF: unigrams + bigrams, 50K features \\
 & Sublinear TF ($\log(1+\text{count})$), IDF weighting \\
Classifier & LinearSVC ($C=1.0$), one-vs-rest \\
Evaluation & 5-fold stratified CV, LOSO, Vulnetic blind test \\
\bottomrule
\end{tabular}
\end{table}

\vspace{0.2em}
\footnotesize\textit{Why this works:} Each LLM family has model-intrinsic command preferences
baked in from training data --- Claude prefers \texttt{python3 -c}, DeepSeek chains with
\texttt{\&\&}, GLM-5 annotates with \texttt{echo}. These lexical choices transfer across scaffolds.
\end{block}

% ── Conclusions ───────────────────────────────────────────────────────────────
\begin{alertblock}{Key Takeaways}
\begin{enumerate}
  \item \textbf{LLMs have behavioral fingerprints} in shell commands that persist across agent scaffolds
  \item \textbf{Evasion-resistant:} mimicry degrades accuracy but does \textbf{not redirect} attribution
  \item \textbf{Cross-model transfer:} fingerprints persist across model tiers ($\sim$70\%)
  \item \textbf{Fingerprinting enables targeted DPI:} per-family routing yields +26.5pp over blind attacks
  \item \textbf{Claude uniquely resistant:} 0\% DPI compliance across all payloads and scaffolds
\end{enumerate}
\end{alertblock}
